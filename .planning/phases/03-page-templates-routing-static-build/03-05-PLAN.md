    Step 1 — datasheet.page.ts:
    ```ts
    @Component({
      selector: 'app-datasheet-page',
      standalone: true,
      changeDetection: ChangeDetectionStrategy.OnPush,
      imports: [PageShellComponent, TwoColumnComponent, HeadingComponent, PinoutComponent, SidenoteComponent, SiteHeaderComponent, SiteFooterComponent, BlockRendererComponent],
      templateUrl: './datasheet.page.html',
      styleUrl: './datasheet.page.scss',
    })
    export class DatasheetPage implements OnInit {
      slug = input.required<string>();
      private api = inject(CONTENT_API);
      private title = inject(Title);
      datasheet = signal<Datasheet | null>(null);

      peripheralBlocks = computed(() => {
        const d = this.datasheet();
        if (!d) return [];
        return (d.peripheralNotes ?? []).filter((b) => b.type !== 'sidenote' && b.type !== 'parts-list');
      });
      peripheralSidenotes = computed(() => {
        const d = this.datasheet();
        if (!d) return [];
        return (d.peripheralNotes ?? [])
          .filter((b): b is Extract<typeof d.peripheralNotes[number], {type:'sidenote'}> => b.type === 'sidenote')
          .sort((a, b) => a.anchorParagraphIndex - b.anchorParagraphIndex);
      });

      async ngOnInit() {
        const d = await this.api.getDatasheet(this.slug());
        this.datasheet.set(d);
        this.title.setTitle(`${d.title} — Arduino UA`);
      }
    }
    ```
