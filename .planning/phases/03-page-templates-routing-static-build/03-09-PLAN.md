    ```js
    #!/usr/bin/env node
    // Phase-exit Lighthouse gate. Runs against the locked target lesson at
    // both desktop and mobile profiles. Hard pass/fail per UI-SPEC §Lighthouse Gates.
    //
    // Thresholds (no tolerance band, per CONTEXT D-LH-03):
    //   LCP < 2.5s   |  CLS < 0.1   |  TBT < 200ms (lab proxy for field INP — Pitfall C)
    //
    // On miss: exit non-zero with remediation guidance per CONTEXT D-LH-04.

    import { spawn } from 'node:child_process';
    import { readFile, writeFile, appendFile } from 'node:fs/promises';
    import { dirname, resolve } from 'node:path';
    import { fileURLToPath } from 'node:url';

    const __dirname = dirname(fileURLToPath(import.meta.url));
    const REPO_ROOT = resolve(__dirname, '..');
    const PORT = 4321;
    const TARGET = `http://localhost:${PORT}/lessons/pershyi-blymayuchyi-svitlodiod/`;
    const THRESHOLDS = { lcp: 2500, cls: 0.1, tbt: 200 };

    function startServer() {
      const proc = spawn('node_modules/.bin/http-server', ['dist/browser', '-p', String(PORT), '--silent'], { cwd: REPO_ROOT, stdio: 'ignore' });
      return proc;
    }

    function waitFor(ms) { return new Promise((r) => setTimeout(r, ms)); }

    async function runLh(profile) {
      const out = `lighthouse-${profile}.json`;
      const args = [
        TARGET,
        '--output=json',
        `--output-path=${out}`,
        `--preset=${profile === 'desktop' ? 'desktop' : 'perf'}`,
        '--quiet',
        '--chrome-flags=--headless=new --no-sandbox',
        '--only-categories=performance',
      ];
      await new Promise((res, rej) => {
        const p = spawn('node_modules/.bin/lighthouse', args, { cwd: REPO_ROOT, stdio: 'inherit' });
        p.on('close', (c) => (c === 0 ? res() : rej(new Error(`lh ${profile} exit ${c}`))));
      });
      const json = JSON.parse(await readFile(resolve(REPO_ROOT, out), 'utf8'));
      const audits = json.audits;
      return {
        profile,
        lcp: audits['largest-contentful-paint'].numericValue,
        cls: audits['cumulative-layout-shift'].numericValue,
        tbt: audits['total-blocking-time'].numericValue,
      };
    }

    function verdict(m) {
      const fails = [];
      if (m.lcp >= THRESHOLDS.lcp) fails.push(`LCP ${m.lcp.toFixed(0)}ms >= ${THRESHOLDS.lcp}ms`);
      if (m.cls >= THRESHOLDS.cls) fails.push(`CLS ${m.cls.toFixed(3)} >= ${THRESHOLDS.cls}`);
      if (m.tbt >= THRESHOLDS.tbt) fails.push(`TBT ${m.tbt.toFixed(0)}ms >= ${THRESHOLDS.tbt}ms`);
      return { pass: fails.length === 0, fails };
    }

    async function main() {
      const server = startServer();
      try {
        await waitFor(1000);
        const desktop = await runLh('desktop');
        const mobile = await runLh('mobile');
        const dv = verdict(desktop); const mv = verdict(mobile);
        const date = new Date().toISOString().slice(0, 10);

        const row = `\n### Phase 3 Performance — ${date}\n\n` +
          `| Profile | LCP | CLS | TBT (INP-proxy) | Verdict |\n` +
          `|---|---|---|---|---|\n` +
          `| desktop | ${desktop.lcp.toFixed(0)}ms | ${desktop.cls.toFixed(3)} | ${desktop.tbt.toFixed(0)}ms | ${dv.pass ? 'PASS' : 'FAIL: ' + dv.fails.join(', ')} |\n` +
          `| mobile  | ${mobile.lcp.toFixed(0)}ms | ${mobile.cls.toFixed(3)} | ${mobile.tbt.toFixed(0)}ms | ${mv.pass ? 'PASS' : 'FAIL: ' + mv.fails.join(', ')} |\n` +
          `\nThresholds: LCP < ${THRESHOLDS.lcp}ms, CLS < ${THRESHOLDS.cls}, TBT < ${THRESHOLDS.tbt}ms (lab proxy for field INP per Pitfall C).\n`;
        await appendFile(resolve(REPO_ROOT, 'docs/typography-checklist.md'), row);

        if (!dv.pass || !mv.pass) {
          console.error('Lighthouse gate FAILED — see docs/typography-checklist.md for details.');
          console.error('Remediation triage (per CONTEXT D-LH-04):');
          console.error('  - LCP miss → verify priority on first figure, font preloads in index.html, woff2 budget.');
          console.error('  - CLS miss → audit unmeasured <img>; verify Fontaine fallback metrics still apply.');
          console.error('  - TBT/INP miss → audit TwoColumn measurement debounce; check Shiki node count.');
          process.exit(1);
        }
        console.log('Lighthouse gate PASS — desktop + mobile both clean.');
      } finally {
        server.kill();
      }
    }

    main().catch((err) => { console.error(err); process.exit(1); });
    ```
