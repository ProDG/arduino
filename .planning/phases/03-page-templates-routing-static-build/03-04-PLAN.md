    ```ts
    import { ChangeDetectionStrategy, Component, computed, inject, input, signal, OnInit } from '@angular/core';
    import { Meta, Title } from '@angular/platform-browser';
    import { RouterLink } from '@angular/router';
    import {
      PageShellComponent, TwoColumnComponent, MarginRailComponent,
      SidenoteComponent, HeadingComponent, LedeComponent,
    } from '@arduino/core-ui';
    import { CONTENT_API } from '../../../content/api/content-api.token';
    import type { Lesson } from '../../../content/models/lesson';
    import { formatDateUk, formatNumberUk } from '../../../lib/intl';
    import { DIFFICULTY_LABELS_UK } from '../../../lib/difficulty';
    import { SiteHeaderComponent } from '../../chrome/site-header.component';
    import { SiteFooterComponent } from '../../chrome/site-footer.component';
    import { BlockRendererComponent } from '../../blocks/block-renderer/block-renderer.component';

    @Component({
      selector: 'app-lesson-page',
      standalone: true,
      changeDetection: ChangeDetectionStrategy.OnPush,
      imports: [
        RouterLink,
        PageShellComponent, TwoColumnComponent, MarginRailComponent,
        SidenoteComponent, HeadingComponent, LedeComponent,
        SiteHeaderComponent, SiteFooterComponent, BlockRendererComponent,
      ],
      templateUrl: './lesson.page.html',
      styleUrl: './lesson.page.scss',
    })
    export class LessonPage implements OnInit {
      slug = input.required<string>();
      private api = inject(CONTENT_API);
      private title = inject(Title);
      private meta = inject(Meta);

      lesson = signal<Lesson | null>(null);
      lessonIndex = signal<{slug:string;title:string}[]>([]);

      bodyBlocks = computed(() => {
        const l = this.lesson();
        if (!l) return [];
        return l.body.filter((b) => b.type !== 'sidenote' && b.type !== 'parts-list');
      });

      sidenotes = computed(() => {
        const l = this.lesson();
        if (!l) return [];
        return l.body
          .filter((b): b is Extract<typeof l.body[number], {type:'sidenote'}> => b.type === 'sidenote')
          .sort((a, b) => a.anchorParagraphIndex - b.anchorParagraphIndex);
      });

      headingToc = computed(() => {
        const l = this.lesson();
        if (!l) return [];
        return l.body
          .filter((b): b is Extract<typeof l.body[number], {type:'heading'}> => b.type === 'heading' && b.level === 2)
          .map((h) => ({ id: h.id ?? '', text: h.text }));
      });

      partsList = computed(() => this.lesson()?.partsList ?? null);

      metaLine = computed(() => {
        const l = this.lesson();
        if (!l) return '';
        const diff = DIFFICULTY_LABELS_UK[l.difficulty];
        const time = formatNumberUk(l.estimatedMinutes);
        const date = formatDateUk(new Date(l.publishedAt));
        return `${diff} · ≈ ${time} хв · ${date}`;
      });

      prevLessonTitle = computed(() => {
        const l = this.lesson();
        return l?.prevSlug ? (this.lessonIndex().find((x) => x.slug === l.prevSlug)?.title ?? '') : '';
      });

      nextLessonTitle = computed(() => {
        const l = this.lesson();
        return l?.nextSlug ? (this.lessonIndex().find((x) => x.slug === l.nextSlug)?.title ?? '') : '';
      });

      async ngOnInit(): Promise<void> {
        const [lesson, index] = await Promise.all([
          this.api.getLesson(this.slug()),
          this.api.listLessons(),
        ]);
        this.lesson.set(lesson);
        this.lessonIndex.set(index.map((l) => ({ slug: l.slug, title: l.title })));
        this.title.setTitle(`${lesson.title} — Arduino UA`);
      }
    }
    ```
