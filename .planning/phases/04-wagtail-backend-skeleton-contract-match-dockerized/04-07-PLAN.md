<objective>
Wire `wagtail-headless-preview` so an editor in the Wagtail admin can click Preview on a draft page and see the unpublished content rendered through the existing Angular `/preview/<contentType>/<token>` route (CSR-only). Authentication is the editor's Django session cookie (D-PREVIEW-01) — the token in the URL is opaque to the FE; same-origin via Traefik means the cookie flows naturally with `credentials: 'include'`. No autosave, no polling — plain browser refresh per D-PREVIEW-03.

Purpose: WAGTAIL-05 mandates headless-preview integration with redirect to the Angular preview route. This plan ships the BE wiring + the FE preview-stub data fetch + a manual editor walkthrough checkpoint.

Output: Updated `pyproject.toml`/`uv.lock` (new dep), `settings/base.py` (INSTALLED_APPS + namespaced dict), `api.py` (endpoint), four page models (mixin), four migration files, `preview-stub.page.ts` (data fetch).
</objective>

       ```typescript
       import { Component, OnInit, inject, input, signal } from '@angular/core';
       import { Meta, Title } from '@angular/platform-browser';
       import { CONTENT_API } from '../../../content/api/content-api.token';
       import { WagtailContentApi } from '../../../content/api/wagtail-content-api';
       import type { Lesson } from '../../../content/models/lesson';
       import type { Article } from '../../../content/models/article';
       import type { Datasheet } from '../../../content/models/datasheet';
       import type { Schematic } from '../../../content/models/schematic';

       type AnyContent = Lesson | Article | Datasheet | Schematic;

       @Component({
         selector: 'app-preview-stub',
         standalone: true,
         templateUrl: './preview-stub.page.html',
         // ... existing imports / styleUrls preserved ...
       })
       export class PreviewStubPage implements OnInit {
         contentType = input.required<string>();
         token = input.required<string>();

         private readonly titleService = inject(Title);
         private readonly meta = inject(Meta);
         private readonly api = inject(CONTENT_API);

         readonly content = signal<AnyContent | null>(null);
         readonly error = signal<string | null>(null);

         async ngOnInit(): Promise<void> {
           this.titleService.setTitle('Попередній перегляд — Arduino UA');
           this.meta.addTag({ name: 'robots', content: 'noindex' });

           if (!(this.api instanceof WagtailContentApi)) {
             this.error.set(
               'Preview requires WagtailContentApi (set environment.useWagtailContentApi=true).'
             );
             return;
           }

           const wagtail = this.api as WagtailContentApi;
           const ct = this.contentType();
           const tok = this.token();
           try {
             // Map FE-side route segment to Wagtail content_type query param.
             // Route is /preview/<contentType>/<token>; contentType is e.g. 'lesson'.
             const map: Record<string, { wagtailType: string; method: keyof WagtailContentApi }> = {
               lesson:    { wagtailType: 'lessons.LessonPage',       method: 'getLessonPreview' },
               article:   { wagtailType: 'articles.ArticlePage',     method: 'getArticlePreview' },
               datasheet: { wagtailType: 'datasheets.DatasheetPage', method: 'getDatasheetPreview' },
               schematic: { wagtailType: 'schematics.SchematicPage', method: 'getSchematicPreview' },
             };
             const entry = map[ct];
             if (!entry) throw new Error(`Unknown preview contentType: ${ct}`);
             const fn = (wagtail as any)[entry.method].bind(wagtail);
             const data = await fn(entry.wagtailType, tok);
             this.content.set(data);
           } catch (err: unknown) {
             this.error.set(err instanceof Error ? err.message : String(err));
           }
         }
       }
       ```
