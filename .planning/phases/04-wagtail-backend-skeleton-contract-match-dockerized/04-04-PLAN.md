    ```python
    """Idempotent contract-test seed command.

    Reads JSON fixtures from src/assets/mock-data/{lessons,articles,datasheets,schematics}/*.json
    (the same files MockContentApi reads) and creates Wagtail page instances with matching
    slugs and StreamField bodies. Seeded pages are PUBLISHED so /api/v2/pages/ surfaces them
    for the contract diff in Plan 06.

    References:
      - D-CONTRACT-03 (the seven fixture slugs)
      - D-CONTRACT-05 (this command's shape — Django management command in apps/contract/)
      - D-MODEL-01..05 (StreamField wire shapes the FE adapter un-wraps)
    """

    from __future__ import annotations

    import io
    import json
    from pathlib import Path
    from typing import Any

    from django.core.files.images import ImageFile
    from django.core.management.base import BaseCommand
    from PIL import Image as PIL
    from wagtail.images.models import Image
    from wagtail.models import Page, Site

    from apps.articles.models import ArticlePage
    from apps.datasheets.models import DatasheetPage
    from apps.lessons.models import LessonPage
    from apps.schematics.models import SchematicPage


    # Path resolution: this file is at backend/apps/contract/management/commands/seed_fixtures.py
    # parents[4] climbs: commands → management → contract → apps → backend
    # Then ../src/assets/mock-data is the FE fixture root.
    MOCK_DATA_ROOT = Path(__file__).resolve().parents[4].parent / "src" / "assets" / "mock-data"

    # D-CONTRACT-03 — exact seven slugs the contract diff exercises.
    LESSON_SLUGS = {
        "pershyi-blymayuchyi-svitlodiod",
        "knopka-ta-pidtyahuvalnyi-rezystor",
        "analogovyi-vhid-ta-potentsiometr",
    }
    ARTICLE_SLUGS = {"chomu-arduino"}
    DATASHEET_SLUGS = {"atmega328p", "arduino-uno-r3"}
    SCHEMATIC_SLUGS = {"blymayuchyi-svitlodiod-shema"}


    def _placeholder_image(title: str, w: int = 800, h: int = 600) -> Image:
        """Create a placeholder Wagtail Image (PNG) so figure/pinout/schematic blocks
        have valid IDs to reference. Idempotent by title.
        """
        existing = Image.objects.filter(title=title).first()
        if existing:
            return existing
        buf = io.BytesIO()
        PIL.new("RGB", (w, h), color=(220, 220, 220)).save(buf, format="PNG")
        buf.seek(0)
        return Image.objects.create(title=title, file=ImageFile(buf, name=f"{title}.png"))


    def _wrap_envelope(block_type: str, value: dict) -> dict:
        """Wagtail StreamField on-disk wire shape: {type, value}. The id field is
        auto-assigned on save. The FE Block contract is FLAT — the FE adapter strips
        this envelope. We DO add it here because Wagtail's StreamField storage layer
        expects it on input.
        """
        return {"type": block_type, "value": value}


    def _block_from_fe(fe_block: dict) -> dict:
        """Convert one flat FE block ({type, ...fields}) into the {type, value: {...}}
        envelope Wagtail StreamField stores. Image references in figure/pinout/
        schematicImage blocks are resolved by uploading a placeholder image and
        substituting the integer ID.
        """
        t = fe_block["type"]
        # shallow-copy without `type`
        value: dict[str, Any] = {k: v for k, v in fe_block.items() if k != "type"}

        if t in {"figure", "pinout"} or t == "schematicImage":
            # FE has flat {src, alt, width, height, ...}; Wagtail expects an `image` integer ID.
            alt = value.get("alt") or "placeholder"
            img = _placeholder_image(
                title=f"seed-{alt[:60]}",
                w=value.get("width") or 800,
                h=value.get("height") or 600,
            )
            value["image"] = img.id
            # Strip FE-only image keys; the BE block reads `image` and re-emits {src, alt, width, height}
            for k in ("src", "width", "height"):
                value.pop(k, None)

        if t == "code":
            # Strip FE `tokens` if present — D-CONTRACT-04 keeps tokens out of BE.
            value.pop("tokens", None)
            # Annotations: each one is FE `{line, html}` flat; Wagtail expects {line, html} StructBlock
            # value. No envelope on inner StructBlock items inside a ListBlock — they're plain dicts.

        if t == "sidenote":
            # FE has `anchorParagraphIndex`; D-MODEL-04 says BE does NOT store it.
            value.pop("anchorParagraphIndex", None)

        return _wrap_envelope(t, value)


    class Command(BaseCommand):
        help = (
            "Idempotently seed contract-test fixtures from src/assets/mock-data into Wagtail "
            "as PUBLISHED pages. Safe to re-run."
        )

        def handle(self, *args, **options):  # noqa: ARG002
            if not MOCK_DATA_ROOT.exists():
                raise SystemExit(
                    f"MOCK_DATA_ROOT does not exist: {MOCK_DATA_ROOT}. "
                    "Did you bind-mount the host repo into the Wagtail container? "
                    "Check compose.dev.yml volumes: ['./backend:/app'] — but src/assets is "
                    "OUTSIDE backend/, so add an additional mount: ./src/assets:/repo/src/assets"
                )

            site = Site.objects.get(is_default_site=True)
            root: Page = site.root_page

            # Step 1: idempotent — delete prior fixture pages by slug
            all_slugs = LESSON_SLUGS | ARTICLE_SLUGS | DATASHEET_SLUGS | SCHEMATIC_SLUGS
            for model in (LessonPage, ArticlePage, DatasheetPage, SchematicPage):
                deleted, _ = model.objects.filter(slug__in=all_slugs).delete()
                if deleted:
                    self.stdout.write(f"  cleared {deleted} prior {model.__name__} fixture pages")

            # Step 2: seed each fixture type
            self._seed_lessons(root)
            self._seed_articles(root)
            self._seed_datasheets(root)
            self._seed_schematics(root)

            self.stdout.write(self.style.SUCCESS("Seed complete."))

        def _seed_lessons(self, root: Page) -> None:
            for path in sorted((MOCK_DATA_ROOT / "lessons").glob("*.json")):
                data = json.loads(path.read_text(encoding="utf-8"))
                if data["slug"] not in LESSON_SLUGS:
                    continue
                lesson = LessonPage(
                    title=data["title"],
                    slug=data["slug"],
                    deck=data.get("deck", ""),
                    difficulty=data.get("difficulty", "beginner"),
                    estimatedMinutes=data.get("estimatedMinutes", 15),
                    body=json.dumps([_block_from_fe(b) for b in data.get("body", [])]),
                    parts_list=json.dumps(
                        [_wrap_envelope("parts_list", {"items": data["partsList"]["items"]})]
                    ),
                )
                if data.get("lede"):
                    lesson.lede = json.dumps(
                        [_wrap_envelope("lede", {"html": data["lede"]["html"]})]
                    )
                root.add_child(instance=lesson)
                lesson.save_revision().publish()
                self.stdout.write(f"  seeded LessonPage: {lesson.slug}")

        def _seed_articles(self, root: Page) -> None:
            for path in sorted((MOCK_DATA_ROOT / "articles").glob("*.json")):
                data = json.loads(path.read_text(encoding="utf-8"))
                if data["slug"] not in ARTICLE_SLUGS:
                    continue
                article = ArticlePage(
                    title=data["title"],
                    slug=data["slug"],
                    deck=data.get("deck", ""),
                    body=json.dumps([_block_from_fe(b) for b in data.get("body", [])]),
                )
                root.add_child(instance=article)
                article.save_revision().publish()
                self.stdout.write(f"  seeded ArticlePage: {article.slug}")

        def _seed_datasheets(self, root: Page) -> None:
            for path in sorted((MOCK_DATA_ROOT / "datasheets").glob("*.json")):
                data = json.loads(path.read_text(encoding="utf-8"))
                if data["slug"] not in DATASHEET_SLUGS:
                    continue
                ds = DatasheetPage(
                    title=data["title"],
                    slug=data["slug"],
                    manufacturer=data.get("manufacturer", ""),
                    pinout=json.dumps([_block_from_fe({"type": "pinout", **data["pinout"]})]),
                    specifications=json.dumps(
                        [_wrap_envelope("spec", {"label": s["label"], "value": s["value"]})
                         for s in data.get("specifications", [])]
                    ),
                    peripheral_notes=json.dumps(
                        [_block_from_fe(b) for b in data.get("peripheralNotes", [])]
                    ),
                )
                root.add_child(instance=ds)
                ds.save_revision().publish()
                self.stdout.write(f"  seeded DatasheetPage: {ds.slug}")

        def _seed_schematics(self, root: Page) -> None:
            for path in sorted((MOCK_DATA_ROOT / "schematics").glob("*.json")):
                data = json.loads(path.read_text(encoding="utf-8"))
                if data["slug"] not in SCHEMATIC_SLUGS:
                    continue
                sch = SchematicPage(
                    title=data["title"],
                    slug=data["slug"],
                    schematic_image=json.dumps(
                        [_block_from_fe({"type": "figure", **data["schematicImage"]})]
                    ),
                    download_url=data.get("downloadUrl", ""),
                    explanation=json.dumps(
                        [_block_from_fe(b) for b in data.get("explanation", [])]
                    ),
                )
                root.add_child(instance=sch)
                sch.save_revision().publish()
                self.stdout.write(f"  seeded SchematicPage: {sch.slug}")
    ```
