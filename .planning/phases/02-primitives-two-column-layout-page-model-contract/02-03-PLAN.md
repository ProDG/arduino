```ts
export interface SidenoteInput {
  /** Top of the anchor element (e.g., `<sup class="sidenote-ref">N</sup>`'s parent paragraph) relative to the grid container, in px. */
  anchorTop: number;
  /** Measured height of the sidenote element in the margin column, in px. */
  height: number;
}

export interface SidenotePlacement {
  /** Final `top` (in px) at which the sidenote should be `position: absolute`-placed. */
  top: number;
}

/**
 * Stack-down collision rule per UI-SPEC ┬¦"Sidenote anchoring mechanism":
 * - Each sidenote's preferred top is its anchor's top.
 * - If sidenote N+1's preferred top is < (sidenote N's top + sidenote N's height + stackGap), slide N+1 down.
 * - Inputs MUST be ordered by anchorTop ascending. Output preserves input order.
 */
export function computeSidenoteStack(
  inputs: ReadonlyArray<SidenoteInput>,
  stackGap: number,
): SidenotePlacement[];
```
